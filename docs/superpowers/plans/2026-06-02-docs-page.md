# Docs Page Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a standalone `/docs` route with sidebar + content layout, referencing ccswitch.io's design pattern, reusing the existing Precision Dark design system.

**Architecture:** Vue 3 SPA with vue-router hash-mode (GitHub Pages compatible). A single `DocsPage.vue` view composes three layout zones — `DocsSidebar` (260px fixed, 5 menu groups + search), central `DocsContent` (markdown rendering + code blocks + callouts), and `DocsTOC` (sticky right-side h2/h3 anchor nav). Documentation content is stored as typed TS objects for now, with markdown strings rendered at build time.

**Tech Stack:** Vue 3 + TypeScript + Vite + vue-router 4 (new dep), zero runtime markdown parser (content pre-structured as Vue component markup)

- [ ] **Step 1: Create DocsContent.vue with markdown parser**

This component takes a raw markdown string and renders it as Vue nodes, detecting code blocks (fenced with ` ``` `), callouts (`:::tip` etc.), tables, and inline code.

```vue
<!-- site/src/components/docs/DocsContent.vue -->
<script setup lang="ts">
import { computed } from 'vue'
import DocCodeBlock from './DocCodeBlock.vue'
import DocCallout from './DocCallout.vue'
import type { DocPage } from '../../types/docs'

const props = defineProps<{
  page: DocPage
  firstPage: boolean
  lastPage: boolean
}>()

/** Simple markdown-to-VNode renderer. Handles code fences, callouts, tables, headings, lists, inline code. */
function parseMarkdown(md: string): Array<{ type: string; [key: string]: any }> {
  const nodes: Array<{ type: string; [key: string]: any }> = []
  const lines = md.split('\n')
  let i = 0

  while (i < lines.length) {
    const line = lines[i]

    // Code fence
    if (line.startsWith('```')) {
      const lang = line.slice(3).trim()
      const codeLines: string[] = []
      i++
      while (i < lines.length && !lines[i].startsWith('```')) {
        codeLines.push(lines[i])
        i++
      }
      i++ // skip closing ```
      nodes.push({ type: 'code', lang: lang || undefined, code: codeLines.join('\n') })
      continue
    }

    // Callout
    const calloutMatch = line.match(/^:::(tip|info|warning)/)
    if (calloutMatch) {
      const calloutType = calloutMatch[1]
      const calloutLines: string[] = []
      i++
      while (i < lines.length && !lines[i].startsWith(':::')) {
        calloutLines.push(lines[i])
        i++
      }
      i++ // skip closing :::
      nodes.push({ type: 'callout', calloutType, content: calloutLines.join('\n') })
      continue
    }

    // Heading
    const h2Match = line.match(/^## (.+)/)
    if (h2Match) {
      const id = h2Match[1].toLowerCase().replace(/[^a-z0-9一-鿿]+/g, '-').replace(/^-|-$/g, '')
      nodes.push({ type: 'h2', id, text: h2Match[1] })
      i++
      continue
    }

    const h3Match = line.match(/^### (.+)/)
    if (h3Match) {
      const id = h3Match[1].toLowerCase().replace(/[^a-z0-9一-鿿]+/g, '-').replace(/^-|-$/g, '')
      nodes.push({ type: 'h3', id, text: h3Match[1] })
      i++
      continue
    }

    // Table — accumulate lines until we hit a non-table line
    if (line.startsWith('|')) {
      const tableLines: string[] = []
      while (i < lines.length && lines[i].startsWith('|')) {
        tableLines.push(lines[i])
        i++
      }
      // Skip separator line (second line like |---|---|)
      const rows = tableLines.filter((_, idx) => idx !== 1 || !tableLines[idx].match(/^\|[\s\-:|]+\|$/))
      nodes.push({ type: 'table', rows: rows.map(r =>
        r.split('|').filter(c => c.trim() !== '').map(c => c.trim())
      )})
      continue
    }

    // Ordered list
    const olMatch = line.match(/^(\d+)\. (.+)/)
    if (olMatch) {
      const items: string[] = []
      while (i < lines.length) {
        const m = lines[i].match(/^(\d+)\. (.+)/)
        if (!m) break
        items.push(m[2])
        i++
      }
      nodes.push({ type: 'ol', items })
      continue
    }

    // Unordered list
    const ulMatch = line.match(/^[-*] (.+)/)
    if (ulMatch) {
      const items: string[] = []
      while (i < lines.length) {
        const m = lines[i].match(/^[-*] (.+)/)
        if (!m) break
        items.push(m[1])
        i++
      }
      nodes.push({ type: 'ul', items })
      continue
    }

    // Blockquote
    if (line.startsWith('> ')) {
      const quoteLines: string[] = []
      while (i < lines.length && lines[i].startsWith('> ')) {
        quoteLines.push(lines[i].slice(2))
        i++
      }
      nodes.push({ type: 'blockquote', text: quoteLines.join('\n') })
      continue
    }

    // Empty line
    if (line.trim() === '') {
      i++
      continue
    }

    // Paragraph
    const paraLines: string[] = []
    while (i < lines.length && lines[i].trim() !== '' && !lines[i].startsWith('```') && !lines[i].startsWith(':::') && !lines[i].startsWith('##') && !lines[i].startsWith('|') && !lines[i].startsWith('>') && !lines[i].match(/^[-*\d]\. /)) {
      paraLines.push(lines[i])
      i++
    }
    if (paraLines.length > 0) {
      nodes.push({ type: 'p', text: paraLines.join(' ') })
    }
  }

  return nodes
}

const parsedNodes = computed(() => parseMarkdown(props.page.content))

/** Render inline markdown: bold, inline code, links */
function renderInline(text: string): string {
  // Inline code
  let html = text.replace(/`([^`]+)`/g, '<code class="inline-code">$1</code>')
  // Bold
  html = html.replace(/\*\*([^*]+)\*\*/g, '<strong>$1</strong>')
  return html
}
</script>

<template>
  <div class="docs-content">
    <!-- Breadcrumb -->
    <div class="breadcrumb">
      <span class="breadcrumb-group">{{ page.groupIcon }} {{ getGroupLabel(page.group) }}</span>
      <span class="breadcrumb-sep">/</span>
      <span class="breadcrumb-current">{{ page.title }}</span>
    </div>

    <!-- Title -->
    <h1 class="content-title">{{ page.title }}</h1>

    <!-- Body -->
    <div class="content-body">
      <template v-for="(node, idx) in parsedNodes" :key="idx">
        <!-- Code block -->
        <DocCodeBlock v-if="node.type === 'code'" :lang="node.lang" :code="node.code" />

interface ParsedNode {
  type: string
  [key: string]: any
}

function parseMarkdown(md: string): ParsedNode[] {
  const nodes: ParsedNode[] = []
  const lines = md.split('\n')
  let i = 0

  while (i < lines.length) {
    const line = lines[i]

    // Code fence
    if (line.startsWith('```')) {
      const lang = line.slice(3).trim()
      const codeLines: string[] = []
      i++
      while (i < lines.length && !lines[i].startsWith('```')) {
        codeLines.push(lines[i])
        i++
      }
      i++
      nodes.push({ type: 'code', lang: lang || undefined, code: codeLines.join('\n') })
      continue
    }

    // Callout :::tip / :::info / :::warning
    const calloutMatch = line.match(/^:::(tip|info|warning)/)
    if (calloutMatch) {
      const calloutType = calloutMatch[1]
      const calloutLines: string[] = []
      i++
      while (i < lines.length && !lines[i].startsWith(':::')) {
        calloutLines.push(lines[i])
        i++
      }
      i++
      nodes.push({ type: 'callout', calloutType, content: calloutLines.join('\n') })
      continue
    }

    // Heading
    const h2Match = line.match(/^## (.+)/)
    if (h2Match) {
      const id = h2Match[1].toLowerCase().replace(/[^a-z0-9一-鿿]+/g, '-').replace(/^-|-$/g, '')
      nodes.push({ type: 'h2', id, text: h2Match[1] })
      i++
      continue
    }

    const h3Match = line.match(/^### (.+)/)
    if (h3Match) {
      const id = h3Match[1].toLowerCase().replace(/[^a-z0-9一-鿿]+/g, '-').replace(/^-|-$/g, '')
      nodes.push({ type: 'h3', id, text: h3Match[1] })
      i++
      continue
    }

    // Table
    if (line.startsWith('|')) {
      const tableLines: string[] = []
      while (i < lines.length && lines[i].startsWith('|')) {
        tableLines.push(lines[i])
        i++
      }
      const rows = tableLines
        .filter((l, idx) => idx !== 1 || !l.match(/^\|[\s\-:|]+\|$/))
        .map(r => r.split('|').filter(c => c !== '').map(c => c.trim()))
      nodes.push({ type: 'table', rows })
      continue
    }

    // Ordered list
    const olMatch = line.match(/^(\d+)\. (.+)/)
    if (olMatch) {
      const items: string[] = []
      while (i < lines.length) {
        const m = lines[i].match(/^(\d+)\. (.+)/)
        if (!m) break
        items.push(m[2])
        i++
      }
      nodes.push({ type: 'ol', items })
      continue
    }

    // Unordered list
    const ulMatch = line.match(/^[-*] (.+)/)
    if (ulMatch) {
      const items: string[] = []
      while (i < lines.length) {
        const m = lines[i].match(/^[-*] (.+)/)
        if (!m) break
        items.push(m[1])
        i++
      }
      nodes.push({ type: 'ul', items })
      continue
    }

    // Blockquote
    if (line.startsWith('> ')) {
      const quoteLines: string[] = []
      while (i < lines.length && lines[i].startsWith('> ')) {
        quoteLines.push(lines[i].slice(2))
        i++
      }
      nodes.push({ type: 'blockquote', text: quoteLines.join('\n') })
      continue
    }

    // Empty line
    if (line.trim() === '') {
      i++
      continue
    }

    // Paragraph
    const paraLines: string[] = []
    while (
      i < lines.length &&
      lines[i].trim() !== '' &&
      !lines[i].startsWith('```') &&
      !lines[i].startsWith(':::') &&
      !lines[i].startsWith('##') &&
      !lines[i].startsWith('|') &&
      !lines[i].startsWith('>') &&
      !lines[i].match(/^[-*\d]\. /)
    ) {
      paraLines.push(lines[i])
      i++
    }
    if (paraLines.length > 0) {
      nodes.push({ type: 'p', text: paraLines.join(' ') })
    }
  }

  return nodes
}

  html = html.replace(/\*\*([^*]+)\*\*/g, '<strong>$1</strong>')
  return html
}
